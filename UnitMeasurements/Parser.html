def parse(string)
  case string
  when COMPLEX_REGEX    then parse_complex(string)
  when SCIENTIFIC_REGEX then parse_scientific(string)
  when RATIONAL_REGEX   then parse_rational(string)
  else                       raise ParseError, string
  end
end
